% Subtitle handling — store separately for professional title page
\makeatletter
\newcommand{\@docsubtitle}{}
\def\subtitle#1{\renewcommand{\@docsubtitle}{#1}}

% Professional title page
\renewcommand{\maketitle}{%
  \begin{titlepage}
    \begin{tikzpicture}[remember picture, overlay]
      % Top colored band
      \fill[blue!70!black] (current page.north west) rectangle ([yshift=-5cm]current page.north east);
      % Accent line below top band
      \fill[blue!40!white] ([yshift=-5cm]current page.north west) rectangle ([yshift=-5.15cm]current page.north east);
      % Bottom colored band
      \fill[blue!70!black] (current page.south west) rectangle ([yshift=1.5cm]current page.south east);
    \end{tikzpicture}
    \vspace*{3cm}
    \begin{center}
      {\fontsize{26}{32}\selectfont\bfseries\color{blue!70!black} \@title \par}
      \vspace{0.7cm}
      {\Large\color{gray!60!black} \@docsubtitle \par}
      \vspace{0.7cm}
      {\large\color{gray!50!black} \@date \quad\textbar\quad \href{https://github.com/fredgis/UDCSP}{\color{blue!60!black}\nolinkurl{github.com/fredgis/UDCSP}} \par}
      \vspace{0.8cm}
    \end{center}
    \noindent\makebox[\textwidth][c]{%
      \hspace*{-2.5cm}\includegraphics[width=\paperwidth,height=11cm,keepaspectratio]{images/Demofull.png}\hspace*{-2.5cm}%
    }
    \vspace*{\fill}
  \end{titlepage}
}
\makeatother

% Screen-shot frame helper — bordered figure for UI captures
\newcommand{\screenfig}[3]{%
  \begin{figure}[H]%
  \centering%
  \setlength{\fboxsep}{0pt}%
  \setlength{\fboxrule}{1.2pt}%
  \fcolorbox{gray!50!black}{white}{\includegraphics[width=#1]{#2}}%
  \caption{#3}%
  \end{figure}%
}

% Persona avatar — small image floated left with paragraph text wrapping
\usepackage{wrapfig}
\newcommand{\persona}[2]{%
  \begin{wrapfigure}{l}{0.26\textwidth}%
    \vspace{-0.4\baselineskip}%
    \centering%
    \includegraphics[width=0.24\textwidth]{#1}%
    \vspace{-1.0\baselineskip}%
  \end{wrapfigure}%
}
